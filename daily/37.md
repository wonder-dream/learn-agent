## 树解题模板

```python
# DFS 递归模板
def dfs(root):
    if not root:
        return  # 或 return 0 / None
    left = dfs(root.left)
    right = dfs(root.right)
    # 处理当前节点
    return combine(left, right)

# BFS 模板
from collections import deque
def bfs(root):
    if not root: return
    q = deque([root])
    while q:
        node = q.popleft()
        # 处理 node
        if node.left: q.append(node.left)
        if node.right: q.append(node.right)

# BST 中序遍历 = 有序序列
def inorder(root):
    if not root: return []
    return inorder(root.left) + [root.val] + inorder(root.right)
```
