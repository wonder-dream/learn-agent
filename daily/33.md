### 1. Trace ID + 上下文 (1.5h)

```python
# src/observability.py
import time
import uuid
import logging
from dataclasses import dataclass, field
from contextvars import ContextVar

# 用 ContextVar 实现请求级别的 Trace ID（异步安全）
trace_id_var: ContextVar[str] = ContextVar('trace_id', default='')

def set_trace(trace_id: str = None):
    trace_id_var.set(trace_id or str(uuid.uuid4())[:8])

def get_trace() -> str:
    return trace_id_var.get()

class TraceFilter(logging.Filter):
    """注入 trace_id 到每条日志"""
    def filter(self, record):
        record.trace_id = get_trace() or 'no-trace'
        return True


@dataclass
class ObservabilityReport:
    trace_id: str = ""
    file_path: str = ""
    file_lines: int = 0

    # 阶段耗时 (ms)
    static_time_ms: float = 0
    extract_time_ms: float = 0
    llm_time_ms: float = 0
    merge_time_ms: float = 0
    total_time_ms: float = 0

    # Token 统计
    input_tokens: int = 0
    output_tokens: int = 0
    total_tokens: int = 0
    llm_calls: int = 0

    # 问题统计
    static_issues: int = 0
    llm_issues: int = 0
    total_issues: int = 0
    false_positives_estimate: int = 0  # 基于历史准确率估算

    # 成本计算 (DeepSeek pricing)
    @property
    def estimated_cost_rmb(self) -> float:
        # DeepSeek pricing (假设)：¥1/百万 input tokens, ¥2/百万 output tokens
        input_cost = self.input_tokens / 1_000_000 * 1.0
        output_cost = self.output_tokens / 1_000_000 * 2.0
        return round(input_cost + output_cost, 6)

    def to_dict(self) -> dict:
        return {
            "trace_id": self.trace_id,
            "file_path": self.file_path,
            "timing": {
                "static_ms": self.static_time_ms,
                "extract_ms": self.extract_time_ms,
                "llm_ms": self.llm_time_ms,
                "merge_ms": self.merge_time_ms,
                "total_ms": self.total_time_ms,
            },
            "tokens": {
                "input": self.input_tokens,
                "output": self.output_tokens,
                "total": self.total_tokens,
                "calls": self.llm_calls,
            },
            "issues": {
                "static": self.static_issues,
                "llm": self.llm_issues,
                "total": self.total_issues,
            },
            "cost_rmb": self.estimated_cost_rmb,
        }


class UsageTracker:
    """全局用量统计（内存版，生产换 DB）"""
    def __init__(self):
        self.total_reviews = 0
        self.total_tokens = 0
        self.total_cost = 0.0
        self.total_issues = 0
        self.start_time = time.time()

    def record(self, report: ObservabilityReport):
        self.total_reviews += 1
        self.total_tokens += report.total_tokens
        self.total_cost += report.estimated_cost_rmb
        self.total_issues += report.total_issues

    def stats(self) -> dict:
        hours = (time.time() - self.start_time) / 3600
        return {
            "total_reviews": self.total_reviews,
            "total_tokens": self.total_tokens,
            "total_cost_rmb": round(self.total_cost, 4),
            "total_issues_found": self.total_issues,
            "avg_issues_per_review": round(self.total_issues / max(self.total_reviews, 1), 1),
            "avg_cost_per_review": round(self.total_cost / max(self.total_reviews, 1), 4),
            "uptime_hours": round(hours, 1),
        }


# 全局单例
_usage_tracker = UsageTracker()

def get_usage_tracker() -> UsageTracker:
    return _usage_tracker


class TraceTimer:
    """计时上下文管理器——自动记录阶段耗时到 report"""
    def __init__(self, report: ObservabilityReport, field: str):
        self.report = report
        self.field = field

    def __enter__(self):
        self.start = time.time()
        return self

    def __exit__(self, *args):
        elapsed = (time.time() - self.start) * 1000
        setattr(self.report, f"{self.field}_time_ms", round(elapsed, 1))
```
