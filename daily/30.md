### 1. MCP 协议理解 (0.5h)

MCP (Model Context Protocol) = 让 LLM 发现和调用工具的标准化协议。

```
┌──────────┐     MCP Protocol      ┌──────────────┐
│  Claude   │ ←──────────────────→  │  MCP Server   │
│  (Host)   │   list_tools         │  (your app)   │
│           │   call_tool          │               │
└──────────┘                       └──────────────┘
```

Server 暴露：
- **Tools**: 可调用的函数（如 review_code）
- **Resources**: 可读取的数据（如项目规范文档）
- **Prompts**: 预定义的 Prompt 模板

```python
# src/mcp_server.py
"""MCP Server for Code Review Agent

启动后，任何支持 MCP 的客户端（Claude Desktop、Cline 等）
都可以自动发现并调用 code_review 工具。
"""

import asyncio
from mcp.server import Server, NotificationOptions
from mcp.server.models import InitializationCapabilities
from mcp.server.stdio import stdio_server
from mcp.types import Tool, TextContent

from src.llm_client import LLMClient
from src.pipeline import ReviewPipeline
from src.schemas import ReviewRequest
import json

# 创建 MCP Server
server = Server("code-review-agent")

# 初始化 LLM（延迟初始化，避免启动时联网）
_llm = None
_pipeline = None

def get_pipeline():
    global _pipeline, _llm
    if _pipeline is None:
        import os
        _llm = LLMClient(
            api_key=os.getenv("DEEPSEEK_API_KEY"),
            base_url=os.getenv("DEEPSEEK_BASE_URL"),
        )
        _pipeline = ReviewPipeline(_llm)
    return _pipeline


@server.list_tools()
async def handle_list_tools() -> list[Tool]:
    """告诉客户端：我能提供这些工具"""
    return [
        Tool(
            name="review_code",
            description="Review Python source code for bugs, style issues, and security vulnerabilities",
            inputSchema={
                "type": "object",
                "properties": {
                    "code": {
                        "type": "string",
                        "description": "Python source code to review"
                    },
                    "file_path": {
                        "type": "string",
                        "description": "File path (for context in review comments)"
                    },
                    "level": {
                        "type": "string",
                        "enum": ["low", "medium", "high"],
                        "default": "medium",
                        "description": "Review depth"
                    }
                },
                "required": ["code"]
            }
        ),
        Tool(
            name="get_review_rules",
            description="Get the list of static analysis rules used by the reviewer",
            inputSchema={
                "type": "object",
                "properties": {}
            }
        ),
    ]


@server.call_tool()
async def handle_call_tool(name: str, arguments: dict) -> list[TextContent]:
    """执行工具调用"""
    if name == "review_code":
        req = ReviewRequest(
            file_path=arguments.get("file_path", "unknown.py"),
            file_content=arguments["code"],
            review_level=arguments.get("level", "medium"),
        )

        pipeline = get_pipeline()
        response, stats = await pipeline.review(req)

        result = {
            "total_issues": response.total_issues,
            "summary": response.summary,
            "issues": [
                {
                    "line": c.line_number,
                    "severity": c.severity.value,
                    "category": c.category,
                    "message": c.message,
                    "suggestion": c.suggestion,
                }
                for c in response.comments
            ],
            "stats": {
                "static_time_ms": round(stats.static_time_ms, 1),
                "llm_calls": stats.llm_calls,
                "llm_time_ms": round(stats.llm_time_ms, 1),
            }
        }

        return [TextContent(
            type="text",
            text=json.dumps(result, indent=2, ensure_ascii=False)
        )]

    elif name == "get_review_rules":
        from src.rules.engine import RuleEngine
        engine = RuleEngine()
        rules = [{"name": r.name, "category": r.category} for r in engine.rules]
        return [TextContent(
            type="text",
            text=json.dumps(rules, indent=2)
        )]

    raise ValueError(f"Unknown tool: {name}")


async def main():
    async with stdio_server() as (read_stream, write_stream):
        await server.run(
            read_stream,
            write_stream,
            InitializationCapabilities(
                sampling={},
                experimental={},
                roots={},
            ),
            notification_options=NotificationOptions(
                prompts_changed=False,
                resources_changed=False,
                tools_changed=True,
            ),
        )

if __name__ == "__main__":
    asyncio.run(main())
```
