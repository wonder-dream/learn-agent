### 1. 异步任务模型 (1h)

代码审查可能耗时 5-30 秒，不能让 HTTP 请求一直等：

```python
# src/server.py
import uuid
from enum import Enum
from fastapi import FastAPI, Depends, HTTPException, BackgroundTasks
from src.dependencies import get_settings, get_llm_client
from src.schemas import ReviewRequest, ReviewResponse
from src.graph import ReviewGraph
from src.middleware import setup_middleware

app = FastAPI(title="Code Review Agent", version="0.2.0")
setup_middleware(app)

# 任务状态存储（生产环境用 Redis，这里用 dict 够了）
class TaskStatus(str, Enum):
    PENDING = "pending"
    RUNNING = "running"
    DONE = "done"
    FAILED = "failed"

tasks: dict[str, dict] = {}

@app.post("/review", response_model=dict)
async def submit_review(
    req: ReviewRequest,
    background: BackgroundTasks,
    llm = Depends(get_llm_client),
):
    """提交审查任务 → 立即返回 task_id → 后台执行"""
    task_id = str(uuid.uuid4())[:8]
    tasks[task_id] = {"status": TaskStatus.PENDING, "result": None, "error": None}

    background.add_task(_run_review, task_id, req, llm)

    return {"task_id": task_id, "status": TaskStatus.PENDING}


async def _run_review(task_id: str, req: ReviewRequest, llm):
    """后台执行审查"""
    try:
        tasks[task_id]["status"] = TaskStatus.RUNNING
        graph = ReviewGraph(llm)
        result = await graph.run(req, thread_id=task_id)

        comments = result.get("all_comments", [])
        # 反序列化 ReviewComment
        from src.schemas import ReviewComment
        review_comments = [ReviewComment(**c) for c in comments]

        tasks[task_id]["result"] = ReviewResponse(
            request_id=task_id,
            comments=review_comments,
            summary=result.get("summary", ""),
            total_issues=len(review_comments),
        )
        tasks[task_id]["status"] = TaskStatus.DONE
    except Exception as e:
        tasks[task_id]["status"] = TaskStatus.FAILED
        tasks[task_id]["error"] = str(e)


@app.get("/review/{task_id}", response_model=ReviewResponse)
async def get_review(task_id: str):
    """查询审查结果"""
    if task_id not in tasks:
        raise HTTPException(status_code=404, detail="Task not found")

    task = tasks[task_id]
    if task["status"] == TaskStatus.PENDING or task["status"] == TaskStatus.RUNNING:
        raise HTTPException(status_code=202, detail=f"Task is {task['status']}")
    elif task["status"] == TaskStatus.FAILED:
        raise HTTPException(status_code=500, detail=task["error"])

    return task["result"]


@app.get("/review/{task_id}/status")
async def get_review_status(task_id: str):
    """查询任务状态（不返回完整结果）"""
    if task_id not in tasks:
        raise HTTPException(status_code=404)

    task = tasks[task_id]
    return {
        "task_id": task_id,
        "status": task["status"],
        "total_issues": task["result"].total_issues if task["result"] else None,
    }


@app.get("/health")
async def health():
    return {"status": "ok", "pending_tasks": sum(1 for t in tasks.values() if t["status"] in ("pending", "running"))}
```

### 2. 测试 API (1h)

用 httpx 或 curl：

```python
# scripts/test_api.py
import httpx
import asyncio

async def main():
    async with httpx.AsyncClient(base_url="http://localhost:8000") as client:
        # 提交任务
        resp = await client.post("/review", json={
            "file_path": "test.py",
            "file_content": "def foo(a,b): return a+b",
            "review_level": "medium",
        })
        task = resp.json()
        print(f"Task: {task}")

        # 轮询状态
        while True:
            status = await client.get(f"/review/{task['task_id']}/status")
            s = status.json()
            print(f"Status: {s['status']}")
            if s['status'] in ('done', 'failed'):
                break
            await asyncio.sleep(1)

        # 获取结果
        if s['status'] == 'done':
            result = await client.get(f"/review/{task['task_id']}")
            review = result.json()
            print(f"Issues: {review['total_issues']}")
            for c in review['comments']:
                print(f"  L{c['line_number']} [{c['severity']}] {c['message']}")

asyncio.run(main())
```
