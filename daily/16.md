### 2. 实现 5 条规则 (2h)

```python
# src/rules/style_rules.py
import ast
import re
from .base import BaseRule, RuleResult
from src.schemas import Severity

class LineLengthRule(BaseRule):
    name = "line_length"
    MAX_LENGTH = 120

    def check(self, code: str, tree) -> list[RuleResult]:
        results = []
        for i, line in enumerate(code.split('\n'), 1):
            if len(line) > self.MAX_LENGTH:
                results.append(RuleResult(
                    line=i,
                    severity=Severity.WARNING,
                    category="style",
                    message=f"Line too long ({len(line)} > {self.MAX_LENGTH})",
                    suggestion="Break into multiple lines",
                ))
        return results


class TrailingWhitespaceRule(BaseRule):
    name = "trailing_whitespace"

    def check(self, code: str, tree) -> list[RuleResult]:
        results = []
        for i, line in enumerate(code.split('\n'), 1):
            if line != line.rstrip():
                results.append(RuleResult(
                    line=i,
                    severity=Severity.INFO,
                    category="style",
                    message="Trailing whitespace",
                ))
        return results


class FunctionLengthRule(BaseRule):
    name = "function_length"
    MAX_LINES = 50

    def check(self, code: str, tree) -> list[RuleResult]:
        results = []
        for node in ast.walk(tree):
            if isinstance(node, ast.FunctionDef):
                func_lines = node.end_lineno - node.lineno + 1
                if func_lines > self.MAX_LINES:
                    results.append(RuleResult(
                        line=node.lineno,
                        severity=Severity.WARNING,
                        category="style",
                        message=f"Function '{node.name}' is too long ({func_lines} lines > {self.MAX_LINES})",
                        suggestion=f"Consider splitting into smaller functions",
                    ))
        return results


class BlankLinesRule(BaseRule):
    name = "blank_lines"
    MAX_BLANK = 2

    def check(self, code: str, tree) -> list[RuleResult]:
        results = []
        lines = code.split('\n')
        blank_count = 0
        for i, line in enumerate(lines, 1):
            if line.strip() == '':
                blank_count += 1
            else:
                if blank_count > self.MAX_BLANK:
                    results.append(RuleResult(
                        line=i - blank_count,
                        severity=Severity.INFO,
                        category="style",
                        message=f"Too many blank lines ({blank_count})",
                        suggestion="Use at most 2 blank lines",
                    ))
                blank_count = 0
        return results


class NoDocstringRule(BaseRule):
    name = "no_docstring"

    def check(self, code: str, tree) -> list[RuleResult]:
        results = []
        for node in ast.walk(tree):
            if isinstance(node, (ast.FunctionDef, ast.ClassDef)):
                if node.name.startswith('_') and node.name != '__init__':
                    continue  # 私有函数不强求
                if not (node.body and isinstance(node.body[0], ast.Expr) and
                        isinstance(node.body[0].value, ast.Constant) and
                        isinstance(node.body[0].value.value, str)):
                    kind = "Function" if isinstance(node, ast.FunctionDef) else "Class"
                    results.append(RuleResult(
                        line=node.lineno,
                        severity=Severity.INFO,
                        category="style",
                        message=f"{kind} '{node.name}' has no docstring",
                    ))
        return results
```

### 3. 规则引擎 (1h)

```python
# src/rules/engine.py
from .style_rules import (
    LineLengthRule, TrailingWhitespaceRule, FunctionLengthRule,
    BlankLinesRule, NoDocstringRule,
)

class RuleEngine:
    def __init__(self):
        self.rules = [
            LineLengthRule(),
            TrailingWhitespaceRule(),
            FunctionLengthRule(),
            BlankLinesRule(),
            NoDocstringRule(),
        ]

    def run_all(self, code: str) -> list:
        import ast
        try:
            tree = ast.parse(code)
        except SyntaxError as e:
            return [{"line": e.lineno or 1, "severity": "error",
                     "category": "syntax", "message": str(e)}]

        results = []
        for rule in self.rules:
            results.extend(rule.check(code, tree))
        return results
```
