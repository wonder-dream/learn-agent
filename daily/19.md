### 1. 候选人提取 (1h)

不能把整个文件扔给 LLM——Token 太多、效果也差。只把"可疑片段"喂给 LLM：

```python
# src/analyzers/candidate_extractor.py
import ast

class CandidateExtractor:
    """从代码中提取 LLM 需要精判的候选片段"""

    def extract(self, code: str, tree: ast.Module) -> list[dict]:
        candidates = []

        for node in ast.walk(tree):
            snippet = self._extract_snippet(code, node)

            if isinstance(node, ast.FunctionDef):
                # 检查函数体内是否有可疑模式
                if self._has_try_without_specific_except(node):
                    candidates.append({
                        "type": "broad_except",
                        "line": node.lineno,
                        "name": node.name,
                        "code": snippet,
                    })
                if self._has_none_return_risk(node):
                    candidates.append({
                        "type": "none_return_risk",
                        "line": node.lineno,
                        "name": node.name,
                        "code": snippet,
                    })
                if self._has_unclosed_resource(node):
                    candidates.append({
                        "type": "resource_leak",
                        "line": node.lineno,
                        "name": node.name,
                        "code": snippet,
                    })
                if self._has_nested_loop_modification(node):
                    candidates.append({
                        "type": "loop_logic",
                        "line": node.lineno,
                        "name": node.name,
                        "code": snippet,
                    })

        return candidates

    def _extract_snippet(self, code: str, node) -> str:
        """从原始代码中切出节点对应的文本片段"""
        lines = code.split('\n')
        if hasattr(node, 'lineno') and hasattr(node, 'end_lineno'):
            snippet_lines = lines[node.lineno - 1:node.end_lineno]
            return '\n'.join(snippet_lines)
        return ""

    def _has_try_without_specific_except(self, func_node) -> bool:
        for node in ast.walk(func_node):
            if isinstance(node, ast.Try):
                for handler in node.handlers:
                    if handler.type is None:
                        return True  # bare except:
        return False

    def _has_none_return_risk(self, func_node) -> bool:
        has_return_none = False
        has_return_value = False
        for node in ast.walk(func_node):
            if isinstance(node, ast.Return):
                if node.value is None:
                    has_return_none = True
                else:
                    has_return_value = True
        return has_return_none and has_return_value  # 混合返回 None 和有值

    def _has_unclosed_resource(self, func_node) -> bool:
        for node in ast.walk(func_node):
            if isinstance(node, ast.Call):
                if isinstance(node.func, ast.Name) and node.func.id == 'open':
                    # 检查 open() 是否在 with 语句内
                    parent = node
                    while hasattr(parent, 'parent'):
                        parent = parent.parent
                        if isinstance(parent, ast.With):
                            return False
                    return True
        return False

    def _has_nested_loop_modification(self, func_node) -> bool:
        # 检查是否在嵌套循环中修改正在迭代的列表
        for outer in ast.walk(func_node):
            if isinstance(outer, (ast.For, ast.While)):
                for inner in ast.walk(outer):
                    if isinstance(inner, (ast.For, ast.While)) and inner != outer:
                        # 有嵌套循环 → 检查是否有列表修改
                        for sub in ast.walk(inner):
                            if isinstance(sub, ast.Call):
                                if isinstance(sub.func, ast.Attribute):
                                    if sub.func.attr in ('append', 'remove', 'pop'):
                                        return True
        return False
```

### 2. LLM 精判 Prompt 调优 (1h)

```python
# src/analyzers/logic_analyzer.py
from src.llm_client import LLMClient, LLMResponse
from src.schemas import ReviewComment, Severity
from src.prompts.system_prompt import CODE_REVIEW_SYSTEM  # 复用 Day 10

LOGIC_CHECK_PROMPT = """审查以下 Python 代码片段，只关注逻辑问题（不要重复风格/命名问题）。

候选问题类型：{candidate_type}

代码：
```python
{code}
```

如果存在逻辑问题，输出 JSON：
{{"has_issue": true, "severity": "error|warning|info", "message": "...", "suggestion": "..."}}
如果没有问题，输出：
{{"has_issue": false}}
"""

class LogicAnalyzer:
    def __init__(self, llm: LLMClient):
        self.llm = llm

    async def analyze_candidate(self, candidate: dict) -> ReviewComment | None:
        prompt = LOGIC_CHECK_PROMPT.format(
            candidate_type=candidate["type"],
            code=candidate["code"],
        )
        response = await self.llm.chat(
            messages=[
                {"role": "system", "content": CODE_REVIEW_SYSTEM},
                {"role": "user", "content": prompt},
            ],
            temperature=0.1,
            max_tokens=512,
        )

        # 解析 LLM 的 JSON 输出
        import json
        try:
            result = json.loads(response.content)
        except json.JSONDecodeError:
            return None  # LLM 输出格式错误，跳过

        if result.get("has_issue"):
            return ReviewComment(
                file_path="",  # 调用方填入
                line_number=candidate["line"],
                severity=result.get("severity", "info"),
                category="logic",
                message=result.get("message", ""),
                suggestion=result.get("suggestion"),
            )
        return None
```

@pytest.fixture
def mock_llm(mocker):

        input_tokens=100, output_tokens=50, model="test",
    )
    return client

@pytest.mark.asyncio
async def test_logic_analyzer_finds_issue(mock_llm):
    analyzer = LogicAnalyzer(mock_llm)
    result = await analyzer.analyze_candidate({
        "type": "none_return_risk",
        "line": 10,
        "code": "def foo():\n    if x: return 1\n    return None",
    })
    assert result is not None
    assert result.category == "logic"
