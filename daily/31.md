## 具体任务

### 1. RAG 核心流程理解 (0.5h)

```
文档 → 文本分块 → Embedding 向量化 → 存入向量数据库
                                          ↓
用户问题 → Embedding 向量化 → 检索相似块 → Prompt 拼接 → LLM 生成答案
```

### 2. 最简实现 (2.5h)

```python
# projects/rag-doc-search/rag.py
"""最简 RAG 系统：用代码审查 Agent 的文档作为知识库"""

import os
import json
import numpy as np
from pathlib import Path
from openai import OpenAI

class SimpleRAG:
    def __init__(self, api_key: str, base_url: str):
        self.client = OpenAI(api_key=api_key, base_url=base_url)
        self.chunks = []         # 文本块
        self.embeddings = []     # 向量

    def ingest_docs(self, docs_dir: str):
        """读取目录下所有 .py 和 .md 文件，分块，向量化"""
        for filepath in Path(docs_dir).rglob("*"):
            if filepath.suffix in ('.py', '.md'):
                text = filepath.read_text(encoding='utf-8')
                chunks = self._chunk(text, max_chars=500)
                self.chunks.extend(chunks)

        # 批量向量化（简化版：逐条调 API）
        for chunk in self.chunks:
            emb = self._embed(chunk)
            self.embeddings.append(emb)

        print(f"Ingested {len(self.chunks)} chunks from {docs_dir}")

    def ask(self, question: str, top_k: int = 3) -> str:
        """检索 + 生成"""
        # 1. 向量化问题
        q_emb = self._embed(question)

        # 2. 计算余弦相似度，取 top_k
        scores = [self._cosine_sim(q_emb, e) for e in self.embeddings]
        top_indices = np.argsort(scores)[-top_k:][::-1]

        # 3. 构建 Prompt
        context = "\n\n---\n\n".join(self.chunks[i] for i in top_indices)
        prompt = f"""Based on the following documentation, answer the question.

Context:
{context}

Question: {question}

Answer:"""

        # 4. 生成答案
        response = self.client.chat.completions.create(
            model=os.getenv("DEEPSEEK_MODEL", "deepseek-v4-pro"),
            messages=[{"role": "user", "content": prompt}],
            temperature=0.3,
            max_tokens=500,
        )
        return response.choices[0].message.content

    def _chunk(self, text: str, max_chars: int = 500) -> list[str]:
        """简单分块：按段落边界 + 大小限制"""
        paragraphs = text.split('\n\n')
        chunks = []
        current = ""
        for para in paragraphs:
            if len(current) + len(para) < max_chars:
                current += para + "\n\n"
            else:
                if current:
                    chunks.append(current.strip())
                current = para + "\n\n"
        if current:
            chunks.append(current.strip())
        return chunks or [text[:max_chars]]

    def _embed(self, text: str) -> list[float]:
        """调用 DeepSeek Embedding API（或用通义千问/智谱的 Embedding）"""
        # 简化版：用文本的前 N 个字符 hash 做模拟向量
        # 实际应调用 Embedding API
        response = self.client.embeddings.create(
            model="text-embedding-3-small",  # DeepSeek 可能不提供 embedding
            input=text[:8000],               # 用 OpenAI 兼容格式
        )
        return response.data[0].embedding

    @staticmethod
    def _cosine_sim(a, b):
        a, b = np.array(a), np.array(b)
        return np.dot(a, b) / (np.linalg.norm(a) * np.linalg.norm(b))


if __name__ == '__main__':
    from dotenv import load_dotenv
    load_dotenv()

    rag = SimpleRAG(
        api_key=os.getenv("DEEPSEEK_API_KEY"),
        base_url=os.getenv("DEEPSEEK_BASE_URL"),
    )

    # 把代码审查 Agent 的源码作为知识库
    rag.ingest_docs("../code-review-agent/src/")

    # 提问
    q = "How does the static analyzer detect naming violations?"
    print(f"Q: {q}")
    print(f"A: {rag.ask(q)}")
```
