```python
# DP 五步法
# 1. 定义 dp[i] 的含义
# 2. 找递推关系（状态转移方程）
# 3. 确定初始值
# 4. 确定遍历顺序
# 5. 返回结果（可能需要遍历 dp 数组找最大值）

# 一维 DP 模板
dp = [0] * n
dp[0] = initial_value

    dp[i] = max(dp[i-1], dp[i-2] + val[i])  # 具体方程看题目
return dp[-1]

# 二维 DP 模板（编辑距离、LCS 等）
dp = [[0] * (m+1) for _ in range(n+1)]
for i in range(1, n+1):
    for j in range(1, m+1):
        if s1[i-1] == s2[j-1]:
            dp[i][j] = dp[i-1][j-1]
        else:
            dp[i][j] = min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1]) + 1
return dp[n][m]
```
