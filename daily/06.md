### 2. 路由 + 参数 (1.5h)

```python
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel

# 路径参数
@app.get("/reviews/{review_id}")
async def get_review(review_id: str):
    return {"review_id": review_id}

# 查询参数
@app.get("/reviews")
async def list_reviews(status: str = "all", limit: int = 10):
    return {"status": status, "limit": limit}

# 请求体（POST）
class CreateReviewRequest(BaseModel):
    file_path: str
    file_content: str
    language: str = "python"

@app.post("/reviews")
async def create_review(req: CreateReviewRequest):
    # req 是 pydantic 自动校验后的 CreateReviewRequest 实例
    return {"id": "fake-id", "file": req.file_path}

# 异常响应
@app.get("/reviews/{review_id}")
async def get_review(review_id: int):
    if review_id < 0:
        raise HTTPException(status_code=400, detail="Invalid ID")
    return {"id": review_id}
```
