```bash
pip install pytest pytest-asyncio pytest-mock
```

```python
import pytest
from src.schemas import ReviewRequest

# 基础断言
def test_valid_review_request():
    req = ReviewRequest(
        file_path="test.py",
        file_content="print('hello')"
    )
    assert req.language == "python"
    assert req.review_level == "medium"  # 默认值

# 异常断言
def test_invalid_line_number():
    from src.schemas import ReviewComment
    from pydantic import ValidationError
    with pytest.raises(ValidationError):
        ReviewComment(
            file_path="x.py",
            line_number=0,      # 无效！
            severity="error",
            category="bug",
            message="test"
        )

# fixture：复用测试数据
@pytest.fixture
def sample_request():
    return ReviewRequest(
        file_path="test.py",
        file_content="def foo():\n    pass\n",
    )

def test_with_fixture(sample_request):
    assert sample_request.file_path == "test.py"

# parametrize：一组输入跑同一测试
@pytest.mark.parametrize("language", ["python", "java", "go"])
def test_languages(language):
    req = ReviewRequest(
        file_path="x." + language,
        file_content="",
        language=language
    )
    assert req.language == language
```

### 2. mock 入门 (0.5h)

```python
# mock 的含义：测试时不真正调用外部服务
def test_review_with_mock_llm(mocker):
    # mock LLM 客户端，让它返回假数据
    mock_client = mocker.patch("src.llm_client.LLMClient.chat")
    mock_client.return_value = "No issues found."

    # 接下来测试审查逻辑，不需要真的调 LLM
    result = review_code("some code")
    assert result is not None
```
