```python
# src/tools/ast_tools.py
import ast
from typing import Optional

class ASTTools:
    """AST 相关工具，封装为可调用的 Tool"""

    @staticmethod
    def parse(code: str) -> dict:
        """解析 Python 代码，返回结构摘要"""
        try:
            tree = ast.parse(code)
        except SyntaxError as e:
            return {"error": str(e), "line": e.lineno}

        functions = []
        classes = []
        imports = []

        for node in ast.walk(tree):
            if isinstance(node, ast.FunctionDef):
                functions.append({
                    "name": node.name,
                    "line": node.lineno,
                    "args": [a.arg for a in node.args.args],
                    "body_length": len(node.body),
                })
            elif isinstance(node, ast.ClassDef):
                methods = [n.name for n in ast.walk(node)
                          if isinstance(n, ast.FunctionDef)]
                classes.append({
                    "name": node.name,
                    "line": node.lineno,
                    "methods": methods,
                })
            elif isinstance(node, ast.Import):
                imports.extend(a.name for a in node.names)
            elif isinstance(node, ast.ImportFrom):
                imports.append(f"{node.module}.{node.names[0].name}")

        return {
            "functions": functions,
            "classes": classes,
            "imports": imports,
            "total_lines": len(code.split('\n')),
        }

    @staticmethod
    def get_function_body(code: str, func_name: str) -> Optional[str]:
        """按函数名提取函数体"""
        tree = ast.parse(code)
        for node in ast.walk(tree):
            if isinstance(node, ast.FunctionDef) and node.name == func_name:
                lines = code.split('\n')
                return '\n'.join(lines[node.lineno-1:node.end_lineno])
        return None
```

### 2. 错误处理 + 重试 (1.5h)

```python
# src/tools/retry.py
import asyncio
import functools
import time
from enum import Enum

class ErrorPolicy(Enum):
    RETRY = "retry"
    SKIP = "skip"
    FAIL = "fail"

class RetryableError(Exception):
    """可重试的错误"""
    pass

class FatalError(Exception):
    """不可恢复的错误——跳过当前 item，继续处理下一个"""
    pass

def with_retry(
    max_retries: int = 3,
    base_delay: float = 1.0,
    max_delay: float = 30.0,
    on_error: ErrorPolicy = ErrorPolicy.RETRY,
):
    """异步重试装饰器，带指数退避和抖动"""
    def decorator(func):
        @functools.wraps(func)
        async def wrapper(*args, **kwargs):
            last_error = None
            for attempt in range(max_retries + 1):
                try:
                    return await func(*args, **kwargs)
                except FatalError:
                    if on_error == ErrorPolicy.SKIP:
                        return None
                    raise
                except Exception as e:
                    last_error = e
                    if attempt == max_retries:
                        if on_error == ErrorPolicy.SKIP:
                            return None
                        raise

                    # 指数退避 + 随机抖动
                    delay = min(base_delay * (2 ** attempt), max_delay)
                    jitter = delay * 0.1 * (hash(str(e)) % 100) / 100
                    await asyncio.sleep(delay + jitter)

            raise last_error
        return wrapper
    return decorator


class TimeoutGuard:
    """超时保护上下文管理器"""
    def __init__(self, seconds: float, message: str = "Operation timed out"):
        self.seconds = seconds
        self.message = message

    async def __aenter__(self):
        self._task = asyncio.current_task()
        self._timer = asyncio.create_task(self._timeout())
        return self

    async def __aexit__(self, *args):
        self._timer.cancel()

    async def _timeout(self):
        await asyncio.sleep(self.seconds)
        if self._task and not self._task.done():
            self._task.cancel()
```

```python
# 在 node_llm_review 中添加
async def node_llm_review(self, state: ReviewState) -> ReviewState:
    candidates = state.get("candidates", [])
    if not candidates:
        return {"llm_comments": []}

    async def analyze_with_timeout(candidate):
        try:
            async with TimeoutGuard(30.0, "LLM call timeout"):
                return await self.logic.analyze_candidate(candidate)
        except asyncio.CancelledError:
            return None  # 超时跳过该候选项

    tasks = [analyze_with_timeout(c) for c in candidates]
    results = await asyncio.gather(*tasks, return_exceptions=True)

    llm_comments = []
    for i, r in enumerate(results):
        if isinstance(r, Exception):
            print(f"Candidate {candidates[i]['type']} failed: {r}")
        elif r is not None:
            llm_comments.append(r.model_dump())

    return {"llm_comments": llm_comments}
```
