### 1. 命名规范规则 (1.5h)

```python
# src/rules/naming_rules.py
import ast
import re
from .base import BaseRule, RuleResult
from src.schemas import Severity

# PEP 8 命名约定
SNAKE_CASE = re.compile(r'^[a-z][a-z0-9_]*$')
CAMEL_CASE = re.compile(r'^[A-Z][a-zA-Z0-9]*$')
UPPER_CASE = re.compile(r'^[A-Z][A-Z0-9_]*$')
# 允许的短变量名（常见惯例）
SHORT_OK = {'x', 'y', 'i', 'j', 'k', 'n', 'f', 's', 'id', 'db', 'os', 'fn'}

class FunctionNamingRule(BaseRule):
    name = "function_naming"
    category = "naming"

    def check(self, code, tree) -> list[RuleResult]:
        results = []
        for node in ast.walk(tree):
            if isinstance(node, ast.FunctionDef):
                if node.name.startswith('_'):
                    continue  # 私有方法跳过
                if not SNAKE_CASE.match(node.name):
                    results.append(RuleResult(
                        line=node.lineno,
                        severity=Severity.WARNING,
                        category="naming",
                        message=f"Function '{node.name}' should use snake_case",
                        suggestion=self._to_snake(node.name),
                    ))
        return results

    @staticmethod
    def _to_snake(name: str) -> str:
        # Simple converter: CamelCase → snake_case
        return re.sub(r'(?<!^)(?=[A-Z])', '_', name).lower()


class VariableNamingRule(BaseRule):
    name = "variable_naming"
    category = "naming"

    def check(self, code, tree) -> list[RuleResult]:
        results = []
        for node in ast.walk(tree):
            if isinstance(node, ast.Name) and isinstance(node.ctx, ast.Store):
                if node.id in SHORT_OK:
                    continue
                if len(node.id) == 1:  # 单字母但不是常见短名
                    results.append(RuleResult(
                        line=node.lineno,
                        severity=Severity.INFO,
                        category="naming",
                        message=f"Single-letter variable '{node.id}' — consider a descriptive name",
                    ))
                elif not SNAKE_CASE.match(node.id) and '_' not in node.id:
                    if not node.id.isupper():  # 常量可能是全大写
                        results.append(RuleResult(
                            line=node.lineno,
                            severity=Severity.INFO,
                            category="naming",
                            message=f"Variable '{node.id}' should use snake_case",
                        ))
        return results


class ClassNamingRule(BaseRule):
    name = "class_naming"
    category = "naming"

    def check(self, code, tree) -> list[RuleResult]:
        results = []
        for node in ast.walk(tree):
            if isinstance(node, ast.ClassDef):
                if not CAMEL_CASE.match(node.name):
                    results.append(RuleResult(
                        line=node.lineno,
                        severity=Severity.WARNING,
                        category="naming",
                        message=f"Class '{node.name}' should use CamelCase",
                    ))
        return results


class ConstantNamingRule(BaseRule):
    name = "constant_naming"
    category = "naming"

    def check(self, code, tree) -> list[RuleResult]:
        results = []
        for node in ast.walk(tree):
            # 模块级全大写变量 → 应视为常量
            if isinstance(node, ast.Assign):
                for target in node.targets:
                    if isinstance(target, ast.Name):
                        if target.id.isupper() and not UPPER_CASE.match(target.id):
                            results.append(RuleResult(
                                line=node.lineno,
                                severity=Severity.INFO,
                                category="naming",
                                message=f"Constant '{target.id}' has unusual naming",
                            ))
        return results
```

### 2. 复杂度规则 (1h)

```python
# src/rules/complexity_rules.py
import ast
from .base import BaseRule, RuleResult
from src.schemas import Severity

class ComplexityRule(BaseRule):
    """圈复杂度 = 分支节点数 + 1"""
    name = "cyclomatic_complexity"
    category = "style"
    MAX_COMPLEXITY = 10
    BRANCH_NODES = (ast.If, ast.For, ast.While, ast.ExceptHandler,
                    ast.And, ast.Or, ast.IfExp)

    def check(self, code, tree) -> list[RuleResult]:
        results = []
        for node in ast.walk(tree):
            if isinstance(node, ast.FunctionDef):
                complexity = 1  # 基线
                for child in ast.walk(node):
                    if isinstance(child, self.BRANCH_NODES):
                        complexity += 1
                    elif isinstance(child, ast.BoolOp):
                        complexity += len(child.values) - 1

                if complexity > self.MAX_COMPLEXITY:
                    results.append(RuleResult(
                        line=node.lineno,
                        severity=Severity.WARNING,
                        category="style",
                        message=f"Function '{node.name}' cyclomatic complexity is {complexity} "
                                f"(max {self.MAX_COMPLEXITY})",
                        suggestion="Break into smaller functions or simplify conditions",
                    ))
        return results


class NestingDepthRule(BaseRule):
    name = "nesting_depth"
    category = "style"
    MAX_DEPTH = 4

    def check(self, code, tree) -> list[RuleResult]:
        results = []
        for node in ast.walk(tree):
            if isinstance(node, ast.FunctionDef):
                depth = self._max_depth(node)
                if depth > self.MAX_DEPTH:
                    results.append(RuleResult(
                        line=node.lineno,
                        severity=Severity.INFO,
                        category="style",
                        message=f"Function '{node.name}' max nesting depth is {depth} "
                                f"(max {self.MAX_DEPTH})",
                        suggestion="Extract inner blocks into separate functions",
                    ))
        return results

    def _max_depth(self, node, current=0):
        if isinstance(node, (ast.If, ast.For, ast.While, ast.Try,
                              ast.With, ast.ExceptHandler)):
            current += 1
        max_d = current
        for child in ast.iter_child_nodes(node):
            d = self._max_depth(child, current)
            max_d = max(max_d, d)
        return max_d
```
