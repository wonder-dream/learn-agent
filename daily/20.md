### 1. 流水线实现 (2.5h)

```python
# src/pipeline.py
import asyncio
import uuid
from dataclasses import dataclass, field
from src.llm_client import LLMClient
from src.static_analyzer import StaticAnalyzer
from src.analyzers.candidate_extractor import CandidateExtractor
from src.analyzers.logic_analyzer import LogicAnalyzer
from src.schemas import ReviewRequest, ReviewResponse, ReviewComment

@dataclass
class PipelineStats:
    static_time_ms: float = 0
    llm_calls: int = 0
    llm_time_ms: float = 0
    total_tokens: int = 0
    candidates_found: int = 0
    issues_confirmed: int = 0

class ReviewPipeline:
    def __init__(self, llm: LLMClient):
        self.static = StaticAnalyzer()
        self.extractor = CandidateExtractor()
        self.logic = LogicAnalyzer(llm)

    async def review(self, req: ReviewRequest) -> tuple[ReviewResponse, PipelineStats]:
        import time
        stats = PipelineStats()

        # 阶段 1：静态分析
        t0 = time.time()
        static_comments = self.static.analyze(req.file_path, req.file_content)
        stats.static_time_ms = (time.time() - t0) * 1000

        # 阶段 2：提取候选代码段（LLM 精判用）
        import ast
        tree = ast.parse(req.file_content)
        candidates = self.extractor.extract(req.file_content, tree)
        stats.candidates_found = len(candidates)

        # 阶段 3：LLM 精判（并发！）
        t0 = time.time()
        llm_tasks = []
        for c in candidates:
            if req.review_level == "low" and c["type"] not in ("resource_leak", "broad_except"):
                continue  # low 级别只检查高风险类型
            llm_tasks.append(self.logic.analyze_candidate(c))
            stats.llm_calls += 1

        llm_results = await asyncio.gather(*llm_tasks)
        stats.llm_time_ms = (time.time() - t0) * 1000

        # 阶段 4：合并 + 去重
        llm_comments = [r for r in llm_results if r is not None]
        stats.issues_confirmed = len(llm_comments)

        all_comments = self._merge(static_comments, llm_comments)

        # 阶段 5：生成摘要
        summary = self._summarize(all_comments, stats)

        response = ReviewResponse(
            request_id=str(uuid.uuid4())[:8],
            comments=all_comments,
            summary=summary,
            total_issues=len(all_comments),
        )

        return response, stats

    def _merge(self, static: list[ReviewComment], llm: list[ReviewComment]
               ) -> list[ReviewComment]:
        # 静态结果不去重，LLM 结果也不去重
        # 但如果 LLM 和静态报告了同一行同一分类的问题，保留 LLM 的（更详细）
        seen = {}
        for c in static:
            seen[(c.line_number, c.category)] = c
        for c in llm:
            key = (c.line_number, c.category)
            if key in seen:
                # 保留 severity 更高的
                if self._severity_rank(c.severity) > self._severity_rank(seen[key].severity):
                    seen[key] = c
            else:
                seen[key] = c
        return sorted(seen.values(), key=lambda c: c.line_number)

    def _severity_rank(self, s) -> int:
        return {"info": 0, "warning": 1, "error": 2, "critical": 3}.get(s, 0)

    def _summarize(self, comments: list[ReviewComment], stats: PipelineStats) -> str:
        errors = sum(1 for c in comments if c.severity in ("error", "critical"))
        warnings = sum(1 for c in comments if c.severity == "warning")
        infos = sum(1 for c in comments if c.severity == "info")
        parts = [f"Found {len(comments)} issues"]
        if errors: parts.append(f"{errors} errors")
        if warnings: parts.append(f"{warnings} warnings")
        if infos: parts.append(f"{infos} info")
        parts.append(f"(static analysis + {stats.llm_calls} LLM calls)")
        return ", ".join(parts)
```

```python
#!/usr/bin/env python
"""端到端测试：用真实代码跑完整流水线"""
import asyncio
import os
from dotenv import load_dotenv
from src.llm_client import LLMClient
from src.pipeline import ReviewPipeline
from src.schemas import ReviewRequest

load_dotenv()

BAD_CODE = '''
import os

def read_config(path):
    f = open(path, 'r')       # 资源泄漏！
    data = f.read()
    if data:
        result = data.upper()
        return result
    return None                # 混合返回 None 和有值

def processItems(items):       # 非 snake_case 函数名
    for item in items:
        try:
            handle(item)
        except:                 # bare except!
            pass
'''

async def main():
    client = LLMClient(
        api_key=os.getenv("DEEPSEEK_API_KEY"),
        base_url=os.getenv("DEEPSEEK_BASE_URL"),
    )
    pipeline = ReviewPipeline(client)

    req = ReviewRequest(
        file_path="test_file.py",
        file_content=BAD_CODE,
        review_level="medium",
    )

    response, stats = await pipeline.review(req)

    print(f"=== Review Results ===")
    print(f"Summary: {response.summary}")
    print(f"Total issues: {response.total_issues}")
    print(f"---")
    for comment in response.comments:
        print(f"  L{comment.line_number} [{comment.severity.value}] "
              f"({comment.category}) {comment.message}")
    print(f"---")
    print(f"Stats: static={stats.static_time_ms:.1f}ms, "
          f"LLM calls={stats.llm_calls}, LLM time={stats.llm_time_ms:.1f}ms")

asyncio.run(main())
```
