### 1. 依赖注入 (1.5h)

```python
from fastapi import Depends, FastAPI
from functools import lru_cache

# 可复用依赖：获取配置
class Settings:
    def __init__(self):
        self.api_key = os.getenv("DEEPSEEK_API_KEY", "")
        self.model = os.getenv("DEEPSEEK_MODEL", "deepseek-v4-pro")

@lru_cache()  # 配置只创建一次
def get_settings() -> Settings:
    return Settings()

# 可复用依赖：获取 LLM 客户端
def get_llm_client(settings: Settings = Depends(get_settings)):
    from src.llm_client import LLMClient
    return LLMClient(api_key=settings.api_key, model=settings.model)

# 依赖使用
@app.post("/review")
async def review_code(
    req: ReviewRequest,
    llm: LLMClient = Depends(get_llm_client),
):
    result = await llm.chat("Review this code")
    return result
```

### 2. 中间件 (1h)

```python
from fastapi import Request
import time
import logging

# 请求日志中间件
@app.middleware("http")
async def log_requests(request: Request, call_next):
    start = time.time()
    response = await call_next(request)
    duration = time.time() - start
    print(f"[{response.status_code}] {request.method} {request.url.path} — {duration:.3f}s")
    return response
```
