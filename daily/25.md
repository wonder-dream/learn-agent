### 2. 共享 fixture (0.5h)

```python
# tests/conftest.py
import pytest
import os
from pathlib import Path

FIXTURES_DIR = Path(__file__).parent / "fixtures"

@pytest.fixture
def bad_code():
    return (FIXTURES_DIR / "bad_code.py").read_text()

@pytest.fixture
def good_code():
    return (FIXTURES_DIR / "good_code.py").read_text()

@pytest.fixture
def mock_llm(mocker):
    """Mock LLM 客户端，避免真实 API 调用"""
    from src.llm_client import LLMClient, LLMResponse
    client = mocker.MagicMock(spec=LLMClient)
    client.chat.return_value = LLMResponse(
        content='{"has_issue": true, "severity": "error", '
                '"message": "Test issue", "suggestion": "Fix it"}',
        input_tokens=50, output_tokens=30, model="mock",
    )
    return client
```

```python
# tests/test_cli.py
import pytest
import asyncio
from cli import CLI

def test_cli_parser_review():
    cli = CLI()
    args = cli.parser.parse_args(['review', 'test.py'])
    assert args.command == 'review'
    assert args.file == 'test.py'
    assert args.level == 'medium'

def test_cli_parser_format():

    args = cli.parser.parse_args(['review', 'test.py', '--format', 'json'])
    assert args.format == 'json'


# tests/test_server.py
import pytest
from httpx import AsyncClient, ASGITransport
from src.server import app
from src.dependencies import get_llm_client

@pytest.fixture
def mock_llm_dep(mock_llm):
    """覆盖 FastAPI 的 LLM 依赖"""
    app.dependency_overrides[get_llm_client] = lambda: mock_llm
    yield
    app.dependency_overrides.clear()

@pytest.mark.asyncio
async def test_health(mock_llm_dep):
    transport = ASGITransport(app=app)
    async with AsyncClient(transport=transport, base_url="http://test") as client:
        resp = await client.get("/health")
        assert resp.status_code == 200
        assert resp.json()["status"] == "ok"

@pytest.mark.asyncio
async def test_submit_review(mock_llm_dep):
    transport = ASGITransport(app=app)
    async with AsyncClient(transport=transport, base_url="http://test") as client:
        resp = await client.post("/review", json={
            "file_path": "test.py",
            "file_content": "def foo(a,b): return a+b",
            "review_level": "low",
        })
        assert resp.status_code == 200
        data = resp.json()
        assert "task_id" in data

    async with AsyncClient(transport=transport, base_url="http://test") as client:
        # Poll until done
        import asyncio
        for _ in range(10):
            status = await client.get(f"/review/{data['task_id']}/status")
            if status.json()["status"] in ("done", "failed"):
                break
            await asyncio.sleep(0.1)
```
