### 2. 调通第一个请求 (1h)

```bash
pip install openai python-dotenv
```

```python
import os
from openai import OpenAI
from dotenv import load_dotenv

load_dotenv()

client = OpenAI(
    api_key=os.getenv("DEEPSEEK_API_KEY"),
    base_url=os.getenv("DEEPSEEK_BASE_URL"),
)

response = client.chat.completions.create(
    model="deepseek-v4-pro",
    messages=[
        {"role": "system", "content": "You are a code reviewer."},
        {"role": "user", "content": "Review this Python code:\n\ndef add(a,b):\n    return a+b"},
        {"role": "assistant", "content": "Found 1 issue: missing spaces after commas in parameter list."},
        {"role": "user", "content": "现在请对上述问题生成结构化 JSON"},
    ],
    # messages 展示了多轮对话格式
    temperature=0.1,   # 审查要精确，低温度
    max_tokens=2048,
)

print(response.choices[0].message.content)
print(f"Tokens used: {response.usage.total_tokens}")
```

**重要理解**：
- `temperature`：越低越确定（审查用 0.1），越高越随机（创意用 0.8+）
- `max_tokens`：限制输出长度，防止超长
- messages 的 role：system（设定人设）、user（用户指令）、assistant（AI 回复）

### 3. 封装 LLMClient (1.5h)

```python
# src/llm_client.py
import os
from typing import Optional
from openai import OpenAI, APIError, RateLimitError
import asyncio
from dataclasses import dataclass

@dataclass
class LLMResponse:
    content: str
    input_tokens: int
    output_tokens: int
    model: str

class LLMClient:
    def __init__(
        self,
        api_key: str,
        base_url: str = "https://api.deepseek.com/v1",
        model: str = "deepseek-v4-pro",
        max_retries: int = 3,
    ):
        self.model = model
        self.client = OpenAI(api_key=api_key, base_url=base_url)
        self.max_retries = max_retries

    async def chat(
        self,
        messages: list[dict],
        temperature: float = 0.1,
        max_tokens: int = 4096,
    ) -> LLMResponse:
        for attempt in range(self.max_retries):
            try:
                resp = self.client.chat.completions.create(
                    model=self.model,
                    messages=messages,
                    temperature=temperature,
                    max_tokens=max_tokens,
                )
                return LLMResponse(
                    content=resp.choices[0].message.content,
                    input_tokens=resp.usage.prompt_tokens,
                    output_tokens=resp.usage.completion_tokens,
                    model=self.model,
                )
            except RateLimitError:
                wait = 2 ** attempt  # 指数退避：1s, 2s, 4s
                print(f"Rate limited, retrying in {wait}s...")
                await asyncio.sleep(wait)
            except APIError as e:
                if attempt == self.max_retries - 1:
                    raise
                await asyncio.sleep(1)

        raise RuntimeError("Max retries exceeded")
```
