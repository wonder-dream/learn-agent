### 1. CLI 实现 (2h)

```python
# cli.py
#!/usr/bin/env python
"""Code Review Agent CLI

Usage:
    python cli.py review <file> [--level low|medium|high] [--output -]
    python cli.py review <file> --format json
    python cli.py review <file> --format markdown > review.md
"""

import asyncio
import argparse
import sys
import os
import json
import time
from pathlib import Path
from dotenv import load_dotenv

load_dotenv()

# 终端颜色
RED = '\033[91m'
YELLOW = '\033[93m'
BLUE = '\033[94m'
GREEN = '\033[92m'
BOLD = '\033[1m'
RESET = '\033[0m'

class CLI:
    def __init__(self):
        self.parser = self._setup_parser()

    def _setup_parser(self):
        parser = argparse.ArgumentParser(
            prog='code-review',
            description='AI-powered code review agent',
        )
        sub = parser.add_subparsers(dest='command')

        # review 子命令
        review = sub.add_parser('review', help='Review a source file')
        review.add_argument('file', help='Path to source file')
        review.add_argument('--level', choices=['low', 'medium', 'high'],
                            default='medium', help='Review depth')
        review.add_argument('--format', choices=['text', 'json', 'markdown'],
                            default='text', help='Output format')
        review.add_argument('--output', '-o', help='Output file (default: stdout)')

        # stats 子命令
        stats = sub.add_parser('stats', help='Show usage statistics')

        return parser

    async def run(self, args=None):
        opts = self.parser.parse_args(args)

        if opts.command == 'review':
            await self._cmd_review(opts)
        elif opts.command == 'stats':
            self._cmd_stats()
        else:
            self.parser.print_help()

    async def _cmd_review(self, opts):
        # 读取文件
        filepath = Path(opts.file)
        if not filepath.exists():
            print(f"Error: File not found: {opts.file}", file=sys.stderr)
            sys.exit(1)

        code = filepath.read_text(encoding='utf-8')

        # 初始化组件
        from src.llm_client import LLMClient
        from src.graph import ReviewGraph
        from src.schemas import ReviewRequest

        client = LLMClient(
            api_key=os.getenv("DEEPSEEK_API_KEY"),
            base_url=os.getenv("DEEPSEEK_BASE_URL"),
        )
        graph = ReviewGraph(client)
        req = ReviewRequest(
            file_path=str(filepath),
            file_content=code,
            review_level=opts.level,
        )

        # 跑审查
        import time
        t0 = time.time()
        print(f"{BLUE}Reviewing {filepath.name} ({len(code)} chars, level={opts.level})...{RESET}")

        result = await graph.run(req)
        elapsed = time.time() - t0

        # 格式化输出
        output = self._format_result(result, opts.format, filepath.name, elapsed)

        # 输出
        if opts.output:
            Path(opts.output).write_text(output, encoding='utf-8')
            print(f"{GREEN}Saved to {opts.output}{RESET}")
        else:
            print(output)

    def _format_result(self, result, fmt, filename, elapsed):
        comments = result.get("all_comments", [])

        if fmt == 'json':
            return json.dumps({
                "file": filename,
                "elapsed": f"{elapsed:.1f}s",
                "total_issues": len(comments),
                "issues": comments,
            }, indent=2, ensure_ascii=False)

        elif fmt == 'markdown':
            lines = [
                f"# Code Review: `{filename}`",
                f"",
                f"**Total issues**: {len(comments)} | **Time**: {elapsed:.1f}s",
                f"",
                "| Line | Severity | Category | Issue |",
                "|------|----------|----------|-------|",
            ]
            for c in comments:
                lines.append(
                    f"| {c.get('line_number', '?')} | "
                    f"{c.get('severity', '?')} | "
                    f"{c.get('category', '?')} | "
                    f"{c.get('message', '?')} |"
                )
            return '\n'.join(lines)

        else:  # text (colored terminal)
            sev_colors = {
                'error': RED, 'critical': RED,
                'warning': YELLOW,
                'info': BLUE,
            }
            lines = [
                f"{BOLD}Review: {filename}{RESET}",
                f"Total: {len(comments)} issues | Time: {elapsed:.1f}s",
                f"{'─' * 60}",
            ]
            for c in comments:
                color = sev_colors.get(c.get('severity', 'info'), '')
                lines.append(
                    f"  L{c.get('line_number', '?'):>4} "
                    f"{color}[{c.get('severity', '?')}]{RESET} "
                    f"({c.get('category', '?')}) "
                    f"{c.get('message', '?')}"
                )
            return '\n'.join(lines)

    def _cmd_stats(self):
        print("Statistics not yet implemented. Coming soon.")


if __name__ == '__main__':
    cli = CLI()
    asyncio.run(cli.run())
```

### 3. 第一次跑自己的代码 (0.5h)

```bash
# 元审查：让 Agent 审查 Agent 自己的代码！
python cli.py review src/graph.py --level high
python cli.py review src/pipeline.py --level high
```

这是验证项目质量的好机会——如果 Agent 在自己代码上查出了真实问题，用 `git diff` 修掉。面试官如果问"你怎么测试这个项目？"，你说"我让它审自己代码，查出了 X 个问题，全修了"——这个回答非常有说服力。
