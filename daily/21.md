## 具体任务

### 1. 定义 ReviewState (0.5h)

```python
# src/graph.py
from typing import TypedDict, Annotated, Optional
import operator
from langgraph.graph import StateGraph, END
from langgraph.checkpoint.memory import MemorySaver
from src.schemas import ReviewComment, ReviewRequest

class ReviewState(TypedDict):
    # 输入
    request: ReviewRequest
    # 阶段输出
    static_comments: list      # JSON-serialized ReviewComment
    candidates: list
    llm_comments: list
    # 最终输出
    all_comments: Annotated[list, operator.add]
    summary: str
    # 控制
    error: Optional[str]
```

### 2. 实现各节点 (2h)

```python
from src.llm_client import LLMClient
from src.static_analyzer import StaticAnalyzer
from src.analyzers.candidate_extractor import CandidateExtractor
from src.analyzers.logic_analyzer import LogicAnalyzer
import ast, json, asyncio

class ReviewGraph:
    def __init__(self, llm: LLMClient):
        self.llm = llm
        self.static = StaticAnalyzer()
        self.extractor = CandidateExtractor()
        self.logic = LogicAnalyzer(llm)
        self.app = self._build()

    # ──── Nodes ────
    async def node_static_analysis(self, state: ReviewState) -> ReviewState:
        """Node 1: 静态分析"""
        req = state["request"]
        comments = self.static.analyze(req.file_path, req.file_content)
        return {
            "static_comments": [c.model_dump() for c in comments],
        }

    async def node_extract_candidates(self, state: ReviewState) -> ReviewState:
        """Node 2: 提取 LLM 精判候选"""
        req = state["request"]
        tree = ast.parse(req.file_content)
        candidates = self.extractor.extract(req.file_content, tree)
        return {"candidates": candidates}

    async def node_llm_review(self, state: ReviewState) -> ReviewState:
        """Node 3: LLM 精判（并发）"""
        candidates = state.get("candidates", [])
        if not candidates:
            return {"llm_comments": []}

        tasks = [self.logic.analyze_candidate(c) for c in candidates]
        results = await asyncio.gather(*tasks)

        llm_comments = [r for r in results if r is not None]
        return {"llm_comments": [c.model_dump() for c in llm_comments]}

    async def node_merge(self, state: ReviewState) -> ReviewState:
        """Node 4: 合并结果"""
        all_comments = state.get("static_comments", []) + state.get("llm_comments", [])
        # 去重
        seen = {}
        for c in all_comments:
            key = (c.get("line_number"), c.get("category"))
            if key not in seen:
                seen[key] = c
            elif self._severity_rank(c.get("severity")) > self._severity_rank(seen[key].get("severity")):
                seen[key] = c

        merged = sorted(seen.values(), key=lambda x: x.get("line_number", 0))
        return {"all_comments": merged}

    async def node_summarize(self, state: ReviewState) -> ReviewState:
        """Node 5: LLM 生成摘要"""
        comments = state.get("all_comments", [])
        error_count = sum(1 for c in comments if c.get("severity") in ("error", "critical"))
        warn_count = sum(1 for c in comments if c.get("severity") == "warning")
        info_count = len(comments) - error_count - warn_count
        return {
            "summary": f"Found {len(comments)} issues: {error_count} errors, {warn_count} warnings, {info_count} info"
        }

    def _severity_rank(self, s) -> int:
        return {"info": 0, "warning": 1, "error": 2, "critical": 3}.get(s, 0)

    # ──── Graph Builder ────
    def _build(self):
        builder = StateGraph(ReviewState)

        builder.add_node("static", self.node_static_analysis)
        builder.add_node("extract", self.node_extract_candidates)
        builder.add_node("llm", self.node_llm_review)
        builder.add_node("merge", self.node_merge)
        builder.add_node("summarize", self.node_summarize)

        builder.set_entry_point("static")
        builder.add_edge("static", "extract")
        builder.add_edge("extract", "llm")
        builder.add_edge("llm", "merge")
        builder.add_edge("merge", "summarize")
        builder.add_edge("summarize", END)

        return builder.compile(checkpointer=MemorySaver())

    async def run(self, request: ReviewRequest, thread_id: str = "default"):
        initial: ReviewState = {
            "request": request,
            "static_comments": [],
            "candidates": [],
            "llm_comments": [],
            "all_comments": [],
            "summary": "",
            "error": None,
        }
        config = {"configurable": {"thread_id": thread_id}}
        result = await self.app.ainvoke(initial, config)
        return result
```

### 3. 为什么 LangGraph 版本更好？(0.5h)

| 维度 | pipeline.py (Day 20) | graph.py (Day 21) |
|------|---------------------|-------------------|
| 流程可见性 | 读代码才能理解 | 图结构一目了然 |
| 错误隔离 | 一处崩全崩 | 节点独立，可单独恢复 |
| 可恢复性 | 无 | Checkpoint 可暂停/恢复 |
| 可测试性 | 只能端到端测 | 每个节点可独立测 |
| 可扩展性 | 加步骤要改主代码 | 加节点 + 连边即可 |
| **面试价值** | 一般 | 高——展示框架理解深度 |
