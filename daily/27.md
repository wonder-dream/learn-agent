```python
# src/logging.py
import logging
import json
import time
import sys
from datetime import datetime, timezone

class JSONFormatter(logging.Formatter):
    def format(self, record):
        log_entry = {
            "timestamp": datetime.now(timezone.utc).isoformat(),
            "level": record.levelname,
            "logger": record.name,
            "message": record.getMessage(),
        }
        if record.exc_info and record.exc_info[0]:
            log_entry["exception"] = self.formatException(record.exc_info)
        # 附加字段
        if hasattr(record, "extra_fields"):
            log_entry.update(record.extra_fields)
        return json.dumps(log_entry, ensure_ascii=False)

def setup_logging(level: str = "INFO"):
    handler = logging.StreamHandler(sys.stdout)
    handler.setFormatter(JSONFormatter())
    root = logging.getLogger()
    root.addHandler(handler)
    root.setLevel(getattr(logging, level.upper()))

# 业务日志辅助
class ReviewLogger:
    def __init__(self, name: str):
        self.logger = logging.getLogger(name)

    def review_start(self, file_path: str, review_level: str):
        self.logger.info("Review started", extra={
            "extra_fields": {"file": file_path, "level": review_level, "event": "review_start"}
        })

    def review_complete(self, file_path: str, issues: int, elapsed_ms: float, tokens: int):
        self.logger.info("Review completed", extra={
            "extra_fields": {
                "file": file_path, "issues": issues,
                "elapsed_ms": round(elapsed_ms, 1), "tokens": tokens,
                "event": "review_complete"
            }
        })

    def review_error(self, file_path: str, error: str):
        self.logger.error(f"Review failed: {error}", extra={
            "extra_fields": {"file": file_path, "event": "review_error"}
        })
```

```python
# src/rate_limiter.py
import asyncio
import time
from collections import deque

class TokenBucket:
    """令牌桶限流器"""
    def __init__(self, rate: int, burst: int):
        self.rate = rate          # tokens per second
        self.burst = burst        # max tokens
        self.tokens = burst
        self.last_refill = time.monotonic()

    async def acquire(self):
        while True:
            now = time.monotonic()
            elapsed = now - self.last_refill
            self.tokens = min(self.burst, self.tokens + elapsed * self.rate)
            self.last_refill = now

            if self.tokens >= 1:
                self.tokens -= 1
                return
            else:
                wait = (1 - self.tokens) / self.rate
                await asyncio.sleep(wait)

from pydantic_settings import BaseSettings
from functools import lru_cache

class Settings(BaseSettings):
    # LLM
    deepseek_api_key: str = ""
    deepseek_base_url: str = "https://api.deepseek.com/v1"
    deepseek_model: str = "deepseek-v4-pro"

    # Server
    host: str = "0.0.0.0"
    port: int = 8000

    # Review limits
    max_file_size_bytes: int = 1_048_576      # 1MB
    max_review_tokens: int = 4096
    llm_timeout_seconds: int = 30
    max_retries: int = 3

    # Rate limit
    llm_rate_per_second: int = 5
    llm_burst: int = 10

    # Logging
    log_level: str = "INFO"

    class Config:
        env_file = ".env"

@lru_cache()
def get_settings() -> Settings:
    return Settings()
```
