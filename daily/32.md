### 1. 混合检索 (1.5h)

```python
# projects/rag-doc-search/hybrid_search.py
from rank_bm25 import BM25Okapi
import numpy as np

class HybridRetriever:
    def __init__(self, chunks: list[str], embeddings: list[np.ndarray]):
        self.chunks = chunks
        self.embeddings = embeddings

        # BM25 索引
        tokenized = [c.lower().split() for c in chunks]
        self.bm25 = BM25Okapi(tokenized)

    def retrieve(self, query: str, top_k: int = 5) -> list[str]:
        # 1. 语义检索
        q_emb = self._embed(query)
        semantic_scores = [self._cosine_sim(q_emb, e) for e in self.embeddings]

        # 2. 关键词检索
        tokenized_q = query.lower().split()
        keyword_scores = self.bm25.get_scores(tokenized_q)

        # 3. 分数融合（归一化 + 加权）
        semantic_norm = self._normalize(semantic_scores)
        keyword_norm = self._normalize(keyword_scores)

        # 语义权重 0.6，关键词权重 0.4
        combined = [0.6 * s + 0.4 * k for s, k in zip(semantic_norm, keyword_norm)]

        # 4. 取 top_k
        top_indices = np.argsort(combined)[-top_k:][::-1]
        return [self.chunks[i] for i in top_indices]

    def _normalize(self, scores):
        min_s, max_s = min(scores), max(scores)
        if max_s == min_s:
            return [0.5] * len(scores)
        return [(s - min_s) / (max_s - min_s) for s in scores]
```

### 2. Rerank (1h)

```python
# projects/rag-doc-search/reranker.py
class LLMReranker:
    """用 LLM 做 cross-encoder 重排序：判断每个文档和 query 的相关性"""
    def __init__(self, client):
        self.client = client

    async def rerank(self, query: str, docs: list[str]) -> list[str]:
        scored = []
        for doc in docs:
            prompt = f"""Rate the relevance of this document to the query on a scale of 1-10.
Query: {query}
Document: {doc[:500]}
Relevance (1-10):"""
            response = await self.client.chat.completions.create(
                model="deepseek-v4-pro",
                messages=[{"role": "user", "content": prompt}],
                temperature=0.1,
                max_tokens=10,
            )
            try:
                score = int(response.choices[0].message.content.strip())
            except ValueError:
                score = 5
            scored.append((score, doc))

        scored.sort(key=lambda x: x[0], reverse=True)
        return [doc for _, doc in scored]
```

```python
# projects/rag-doc-search/eval.py
TEST_QUESTIONS = [
    {
        "question": "What static analysis rules exist?",
        "expected_keywords": ["LineLength", "FunctionLength", "NoDocstring", "naming"],
    },
    {
        "question": "How does the LLM deep analysis work?",
        "expected_keywords": ["candidate", "extract", "LogicAnalyzer", "prompt"],
    },
    # ... 再 3 个
]

def evaluate(retriever, questions):
    for q in TEST_QUESTIONS:
        docs = retriever.retrieve(q["question"], top_k=5)
        combined_text = " ".join(docs).lower()
        hits = sum(1 for kw in q["expected_keywords"] if kw.lower() in combined_text)
        recall = hits / len(q["expected_keywords"])
        print(f"Q: {q['question'][:50]}... → Recall: {recall:.0%}")
```
