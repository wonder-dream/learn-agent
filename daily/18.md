```python
# src/rules/converter.py
from .base import RuleResult
from src.schemas import ReviewComment, Severity

def to_review_comment(rule_result: RuleResult, file_path: str) -> ReviewComment:
    severity_map = {
        "info": Severity.INFO,
        "warning": Severity.WARNING,
        "error": Severity.ERROR,
        # 兼容 RuleResult 中的大写
        "INFO": Severity.INFO,
        "WARNING": Severity.WARNING,
        "ERROR": Severity.ERROR,
    }
    return ReviewComment(
        file_path=file_path,
        line_number=rule_result.line,
        severity=severity_map.get(rule_result.severity, Severity.INFO),
        category=rule_result.category,
        message=rule_result.message,
        suggestion=rule_result.suggestion,
    )
```

### 2. 静态分析入口 (0.5h)

```python
# src/static_analyzer.py
import ast
from src.rules.engine import RuleEngine
from src.rules.converter import to_review_comment
from src.schemas import ReviewComment

class StaticAnalyzer:
    def __init__(self):
        self.engine = RuleEngine()

    def analyze(self, file_path: str, code: str) -> list[ReviewComment]:
        try:
            tree = ast.parse(code)
        except SyntaxError as e:
            return [ReviewComment(
                file_path=file_path,
                line_number=e.lineno or 1,
                severity=Severity.ERROR,
                category="syntax",
                message=f"Syntax error: {e.msg}",
            )]

        raw_results = self.engine.run_all(code)

        # 去重（同一行同一 category 只保留最高 severity）
        seen = {}
        for r in raw_results:
            key = (r.line, r.category)
            if key not in seen or severity_rank(r.severity) > severity_rank(seen[key].severity):
                seen[key] = r

        return [to_review_comment(r, file_path) for r in seen.values()]
```

### 3. 集成测试 (1.5h)

写 `tests/test_static_analyzer.py`：

```python
import pytest
from src.static_analyzer import StaticAnalyzer

SAMPLE_BAD_CODE = '''
import os, sys, json   # 多import在一行

def MyFunction(x,y,z):
    # 参数缺少空格、命名非snake_case
    result = x+y+z
    if result > 0:
        if result > 10:
            if result > 100:
                if result > 1000:
                    if result > 10000:
                        print("Deeply nested!")
    return result

class my_class:  # 非CamelCase
    def __init__(self):
        self.VeryLongVariableNameThatExceedsNormalConvention = 1
'''

def test_static_analyzer_finds_issues():
    analyzer = StaticAnalyzer()
    comments = analyzer.analyze("test.py", SAMPLE_BAD_CODE)
    assert len(comments) > 0, "Should find at least one issue"

    categories = {c.category for c in comments}
    assert "naming" in categories, "Should find naming issues"
    assert "style" in categories, "Should find style issues"

def test_static_analyzer_good_code():
    analyzer = StaticAnalyzer()
    comments = analyzer.analyze("good.py", GOOD_CODE)
    # 好代码应该只有少量或零 info 级别问题
    real_issues = [c for c in comments if c.severity != "info"]
    assert len(real_issues) == 0

def test_syntax_error():
    analyzer = StaticAnalyzer()
    comments = analyzer.analyze("broken.py", "def foo(:")
    assert any(c.category == "syntax" for c in comments)

def test_performance():
    import time
    analyzer = StaticAnalyzer()
    start = time.time()
    for _ in range(100):
        analyzer.analyze("test.py", SAMPLE_BAD_CODE)
    avg = (time.time() - start) / 100
    assert avg < 0.1, f"Static analysis too slow: {avg*1000:.1f}ms (target < 100ms)"
```
